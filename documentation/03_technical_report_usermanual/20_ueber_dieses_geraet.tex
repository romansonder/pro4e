\begin{itemize}
\item Computer mit USB 2.0 Schnittstelle
\end{itemize}
\begin{itemize}
\item Windows 7 oder höher
\end{itemize}
\begin{itemize}
\item Java runtime environment (JRE) installiert
\end{itemize}
